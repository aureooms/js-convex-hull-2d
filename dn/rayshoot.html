Computes the hyperplane in Ax = b closest to z with respect to the direction r. Complexity is O(nd).

Parameters:

  • d must be a dimension.
    (space dimension)

  • m must be a count.
    (the number of hyperplanes)

  • A must be a matrix.
    (an m x d matrix)

  • b must be a vector.
    (a vector of length d)

  • z must be a vertex.
    (the point to shoot from)

  • r must be a direction.
    (the direction to shoot in)

let rayshoot = function ( d , m , A , b , z , r ) { let i = 0 ; let j = m ; let w ; while ( true ) { if ( i >= j ) return j ; let h = A[i] ; w = ( b[i] - vdot( d , h , z ) ) / vdot( d , h , r ) ; if ( w >= 0 ) break ; ++i ; } let k = i ; while ( true ) { ++i ; if ( i >= j ) return k ; let h = A[i] ; let l = ( b[i] - vdot( d , h , z ) ) / vdot( d , h , r ) ; if ( l < 0 || l >= w ) continue ; k = i ; w = l ; } } ; exports.rayshoot = rayshoot ;
